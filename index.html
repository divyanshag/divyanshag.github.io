// branch_ranking.sql
-- MTD CASA growth + rank per region
SELECT branch_id, region,
  casa_balance,
  RANK() OVER (
    PARTITION BY region
    ORDER BY casa_growth_pct DESC
  ) AS region_rank

-- Flag accounts at delinquency risk
SELECT account_id, dpd_bucket,
  missed_emi_count,
  CASE WHEN missed_emi_count >= 2
    AND bal_drop_pct > 40
    THEN 'HIGH'
    ELSE 'WATCH' END AS risk_flag
FROM vw_loan_health
WHERE is_active = 1;

  source_total, gl_total,
  source_total - gl_total AS diff
